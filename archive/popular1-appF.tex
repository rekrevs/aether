\subsection*{Key Relationships}

\textbf{Power flow (Eq. 6.1):}
\[
P_{\sigma}(e) = \varepsilon\,K(e)\,\mathcal{Q}(e,t)\,\tilde{\Delta\Phi}(e)\,P_{\rm pump}(e)
\]

\textbf{Bitrate bound (Eq. 8.3):}
\[
R_{bit} \leq \beta \, \frac{P_{pump}}{k_B T \ln 2} \, \mathcal{Q} \, e^{-d_\sigma/\lambda_\sigma}
\]

\textbf{Anisotropy constraint (Eq. 11.4):}
\[
\varepsilon_\gamma\left(\frac{\lambda_\sigma}{\lambda_C}\right)\mathcal{Q}_\gamma \;\lesssim\; 10^{-18}
\]

\textbf{Sidereal amplitude (matter sector, Eq. 12.4):}
\[
A_{\rm sid}^{(\mathrm{mat})} \;\simeq\;
\frac{\varepsilon_{\rm mat}}{\varepsilon_\gamma} \cdot \frac{\lambda_\sigma}{L_{\rm exp}} \cdot \frac{\mathcal{Q}_{\rm mat}}{\mathcal{Q}_\gamma} \cdot \Xi
\]

\subsection*{Usage Notes}

\begin{itemize}
\item \textbf{Sector separation:} Photon-sector parameters ($\varepsilon_\gamma, \mathcal{Q}_\gamma$) and matter-sector parameters ($\varepsilon_{\rm mat}, \mathcal{Q}_{\rm mat}$) can differ by many orders of magnitude. Optical bounds constrain the former; E1/E2 experiments probe the latter.

\item \textbf{Efficiency $\beta$:} Accounts for losses and saturation effects in converting pump power to effective substrate coupling. Not to be confused with $\beta = 1/(k_B T)$ from statistical mechanics.

\item \textbf{Quality factor $\mathcal{Q}$:} Measures coherence and resonance quality. Experimentally, aim for $\mathcal{Q} \sim 10^{-8}$ (baseline) to $10^{-2}$ (ambitious near criticality).

\item \textbf{Similarity kernel $K$:} Falls exponentially with structural distance. Perfect match: $K = 1$; large mismatch: $K \to 0$.
\end{itemize}
